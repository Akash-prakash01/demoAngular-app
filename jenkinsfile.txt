pipeline {
    agent any
    tools {
        nodejs 'nodeJS'
    }
    environment {
        ANGULAR_CLI_VERSION = '21.2.5'
        NODE_ENV = 'Production'
        CHROME_BIN = 'C:\\Program Files\\Google\\Chrome\\Application'
        DISPLAY = ':99'
    }
    stages {
        stage('Checkout') {
            steps {
                echo 'Checkout Source Code'
                checkout scm
                script {
                    echo "Build No. : ${env.BUILD_NUMBER}"
                    echo "Branch : ${env.BRANCH_NAME}"
                    echo "Commit : ${env.GIT_COMMIT}"
                }
            }
        }

        stage('Install Dependencies') {
            steps {
                echo 'Installing NodeJS Dependencies'
                sh '''
                    node --version
                    npm --version
                    npm ci --only=production
                '''
            }
        }

        stage('Code Quality and Linting') {
            parallel {
                stage('esLint') {
                    steps {
                        echo 'Running esLint'
                        sh 'npm run lint'
                        publishHTML([
                            allowMissing: false,
                            alwaysLinkToLastBuild: true,
                            keepAll: true,
                            reportDir: 'lint-results',
                            reportFiles: 'eslint-report.html',
                            reportName: 'eslint report'
                        ])
                    }
                }
                stage('Prettier Check') {
                    steps {
                        echo 'Check Code Formatting'
                        sh 'npm run prettier:check'
                    }
                }
                stage('Security Check') {
                    steps {
                        echo 'Check Security'
                        sh 'npm audit --audit-level=high'
                    }
                }
                stage('Unit Test') {
                    steps {
                        echo 'Unit Test Check'
                        sh '''
                            export DISPLAY=:99
                            npm run test:ci
                        '''
                    }
                }
                stage('Build') {
                    steps {
                        echo 'Building Angular Application'
                        script {
                            def buildCommand
                            if (env.BRANCH_NAME == 'main' || env.BRANCH_NAME == 'master') {
                                buildCommand = 'npm run build:prod'
                            } else if (env.BRANCH_NAME == 'dev') {
                                buildCommand = 'npm run build:staging'
                            } else {
                                buildCommand = 'npm run build'
                            }
                            sh buildCommand
                        }
                    }
                }
                stage('End to End Test') {
                    when {
                        anyOf {
                            branch 'main'
                            branch 'master'
                            branch 'dev'
                        }
                    }
                    steps {
                        echo 'Running end to end test'
                        sh '''
                            npm run start:ci &
                            server_pid=$!
                            sleep 30
                            npm run e2e:ci
                            kill $server_pid
                        '''
                    }
                }
            }
        }
    }
}
